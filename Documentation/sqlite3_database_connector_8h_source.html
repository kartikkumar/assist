73  Sqlite3DatabaseConnector( const std::string& absolutePathToDatabase )
74  {
75  if ( sqlite3_open( absolutePathToDatabase.c_str( ), &database ) != SQLITE_OK )
76  {
77  closeDatabase( );
78  boost::throw_exception(
79  boost::enable_error_info(
80  std::runtime_error( "Database could not be opened!" ) ) );
81  }
82  }
83 
85  ~Sqlite3DatabaseConnector( ) { closeDatabase( ); }
86 
88  void closeDatabase( ) { sqlite3_close( database ); }
89 
90  void execute( const std::string& sqlCommand )
91  {
92  // Check if sql statement exited with an error message and throw runtime error if necessary.
93  if ( sqlite3_exec( database, sqlCommand.c_str( ), NULL, 0, &databaseErrorMessage )
94  != SQLITE_OK )
95  {
96  sqlite3_free( databaseErrorMessage );
97  closeDatabase( );
98  boost::throw_exception( boost::enable_error_info(
99  std::runtime_error( "Statement execution failed!" ) ) );
100  }
101  }
102 
103  void prepare_v2( const std::string& sqlCommand )
104  {
105  if ( sqlite3_prepare_v2( database, sqlCommand.c_str( ),
106  sqlCommand.size( ), &sqlStatement, 0 ) != SQLITE_OK )
107  {
108  finalizeStatement( );
109  closeDatabase( );
110  boost::throw_exception(
111  boost::enable_error_info(
112  std::runtime_error( "SQL statement was not prepared correctly!" ) ) );
113  }
114  }

146  int step( )
147  {
148  int status = sqlite3_step( sqlStatement );
149 
150  if ( status == SQLITE_ERROR )
151  {
152  finalizeStatement( );
